### 4.20

skill没有上传的必要性，有几个问题在B/S架构难以解决，skill本质上是一个文件夹，一个文件夹就是一个skill。
如果添加skill功能的话，第一个问题skill放在哪？服务器？但是问题在于..
看了下大厂的实现，我大概有思路，然后了我看了一下午大厂是怎么搞得，我看的MinMax，把表设计出来了，然后优化了下获取聊天记录
可以获取元数据了，以及之前没有处理好，多模态聊天，导致会出现key相同覆盖的情况，也就是说看似上传了多个文件，但是获取聊天记录一看，只有最后的文件是存在的，之前的文件都没了。
然后修改为存储一个集合，就解决了这个问题了。
表设计如下
```sql
-- mcp
create table mcp_information(
      id bigint primary key ,
      name varchar(255) not null ,
      url varchar(255) not null ,
      description varchar(500),
      header jsonb default '{}',
      user_id bigint not null ,
      constraint fk_user foreign key (user_id) references "user"(id) on delete cascade
);
comment on table mcp_information is 'MCP配置信息';

-- 目录结构：用户ID/skillID/文件压缩包
-- skill
create table skill_information(
    id bigint primary key,
    source_path varchar(255) not null ,
    skill_name varchar(255) not null ,
    description varchar(1024),
    author bigint not null ,
    use_count int default 0,
    is_public bool default false,
    is_system bool default false,
    create_time timestamptz default current_timestamp,
    update_time timestamptz default current_timestamp
);
comment on table skill_information is 'SKILL配置信息';

-- 用户配置表

create table user_config(
    user_id bigint primary key,
    llm_api_token jsonb default '{}',
    mcp_token jsonb default '{}',
    user_default jsonb default '{}',
    constraint fk_user foreign key (user_id) references "user"(id) on delete cascade
);
comment on table user_config is '用户配置信息';

create table user_skill(
    user_id bigint,
    skill_id bigint,
    create_time timestamptz default current_timestamp,
    primary key (user_id,skill_id),
    constraint fk_user foreign key (user_id) references "user"(id) on delete cascade ,
    constraint fk_skill foreign key (skill_id) references skill_information(id) on delete cascade
);

comment on table user_config is '用户SKILL关系模型';

```
